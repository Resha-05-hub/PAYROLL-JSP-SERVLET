Table Name: PAYROLL_TB

CREATE TABLE PAYROLL_TB (
    RECORDID      VARCHAR2(12 BYTE) PRIMARY KEY,
    EMPLOYEENAME  VARCHAR2(40 BYTE) NOT NULL,
    DESIGNATION   VARCHAR2(30 BYTE) NOT NULL,
    PAYMENT_DATE  DATE NOT NULL,
    SALARY        NUMBER,
    DEPARTMENT    VARCHAR2(30 BYTE),
    REMARKS       VARCHAR2(100 BYTE)
);

Sequence Name: PAYROLL_SEQ

CREATE SEQUENCE PAYROLL_SEQ
START WITH 10
INCREMENT BY 1
MAXVALUE 99;

SELECT * FROM PAYROLL_TB;
SELECT PAYROLL_SEQ.NEXTVAL FROM DUAL;
